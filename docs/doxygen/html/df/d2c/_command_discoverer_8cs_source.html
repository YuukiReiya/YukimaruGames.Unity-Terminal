47 public IEnumerable<CommandSpecification> Discover(string assemblyName)
48 {
49 AssemblyName referencedAssemblyName = null;
50
51 try
52 {
53 referencedAssemblyName = new AssemblyName(assemblyName);
54 var assembly = Assembly.Load(assemblyName);
55 if (assembly is null)
56 {
57 _logger?.Send(MessageType.Error, $"Failed to load assembly: {assemblyName}. Assembly.Load returned null.");
58 return Enumerable.Empty<CommandSpecification>();
59 }
60
61 var dic = new Dictionary<AssemblyName, Assembly>(new[] { new KeyValuePair<AssemblyName, Assembly>(referencedAssemblyName, assembly) });
62 var referencedAssembliesNames = assembly.GetReferencedAssemblies();
63 foreach (var name in referencedAssembliesNames)
64 {
65 if (!dic.TryGetValue(name, out assembly))
66 {
67 assembly = Assembly.Load(name);
68 if (assembly != null)
69 {
70 dic.Add(name, assembly);
71 }
72 }
73 }
74
75 var specs = new List<CommandSpecification>();
76 foreach (var kvp in dic)
77 {
78 assembly = kvp.Value;
79 foreach (var type in GetTypesSafely(assembly))
80 {
81 foreach (var method in GetMethodsSafely(type, kBindingFlags))
82 {
83 if (!TryGetAttribute(method, out var attribute))
84 {
85 continue;
86 }
87
88 if (!IsDiscoverable(method, attribute))
89 {
90 continue;
91 }
92
93 specs.Add(new CommandSpecification(method, attribute.Meta));
94 }
95 }
96 }
97
98 return specs;
99 }
100 catch (Exception e)
101 {
102 _logger?.Send(
103 MessageType.Exception,
104 $"Referenced assembly '{referencedAssemblyName}' from assembly '{assemblyName}' could not be loaded: {e.GetType()}{Environment.NewLine}{e.Message}");
105 throw;
106 }
107 }

175 private bool IsDiscoverable(MethodInfo methodInfo, RegisterAttribute attribute)
176 {
177 // TODO:
178 // kBindingFlagsでInstanceメソッドを取りのぞいているので基本的には通らないはず.
179 if (!methodInfo.IsStatic)
180 {
181 _logger?.Send(

184 return false;
185 }
186
187 var commandName = attribute.Meta.Command;
188 if (string.IsNullOrWhiteSpace(commandName))
189 {
190 _logger?.Send(
191 MessageType.Warning,

193 return false;
