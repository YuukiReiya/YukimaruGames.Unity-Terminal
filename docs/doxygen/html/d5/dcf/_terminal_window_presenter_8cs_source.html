95 {
96 _cts?.Cancel();
97 _cts?.Dispose();
98
99 if (Mathf.Approximately(0f, Duration))
100 {
101 _cts?.Dispose();
102 _cts = null;
103 IsAnimating = false;
104 Evaluate(0f, 0f);
105 _onCompleted?.Invoke(State);
106 return;
107 }
108
109 _cts = new CancellationTokenSource();
110 IsAnimating = true;
111
112 try
113 {
114 var duration = Duration * Scale;
115 var token = _cts.Token;
116 var elapsedTime = 0f;
117 while (elapsedTime < duration)
118 {
119 token.ThrowIfCancellationRequested();
120 Evaluate(duration, elapsedTime);
121 await Task.Yield();
122 elapsedTime += Time.deltaTime;
123 }
124
125 Evaluate(duration, duration);
126 _onCompleted?.Invoke(State);
127 }
128 catch (OperationCanceledException)

130 _onAborted?.Invoke(State);
131 }
132 finally
133 {
134 IsAnimating = false;
