110 void ITerminalService.Execute(string str)
111 {
112 _logger?.Send(MessageType.Entry, str);
113 _history.Add(str);
114
115 var result = _parser.Parse(str, out var parse);
116
117 if (string.IsNullOrEmpty(parse.Command))
118 {
119 return;
120 }
121
122 if (!_registry.TryGet(parse.Command, out var handler))
123 {
124 _logger?.Send(MessageType.Error, $"No such command: '{parse.Command}'.");
125 return;
126 }
127
128 // 構文エラー.
129 if (0 < (result & ICommandParser.ParseStatusCode.SyntaxError))
130 {
131 _logger?.Send(
132 MessageType.Error,
133 $"Invalid string format: \"{str}\" is not enclosed with single (\') or double (\") quotes.");
134 }
135
136 try
137 {
138 _invoker.Execute(handler, parse.Arguments);
139 }
140 catch (CommandArgumentException e)
141 {
142 _logger?.Send(MessageType.Exception, $"Error: {e.Message}");
143 }
144 catch (CommandFormatException e)
145 {
146 _logger?.Send(MessageType.Exception, $"Error: {e.Message}");
147 }
148 catch (Exception e)
149 {
150 _logger?.Send(MessageType.Exception, $"{e.GetType().Name}: {e.Message}");
151 }
152 }
