173 private void Add(Type declaringType, MethodInfo method)
174 {
175 try
176 {
177 if (System.Attribute.GetCustomAttribute(method, typeof(RegisterAttribute)) is not RegisterAttribute
178 attribute)
179 {
180 // 属性が付与されていないメソッドはスキップ.
181 return;
182 }
183
184 if (!method.IsStatic)
185 {
186 // Assemblyから登録するときは呼び出し元のインスタンスが特定できないはずなのでstaticのみ.
187 _logger?.Send(
188 MessageType.Warning,
189 $"Skipping non-static method '{method.Name}' in type '{declaringType.FullName}'. Only static methods can be registered.");
190 return;
191 }
192
193 var command = attribute.Meta.Command;
194 if (string.IsNullOrEmpty(command) || string.IsNullOrWhiteSpace(command))
195 {
196 _logger?.Send(
197 MessageType.Warning,
198 $"Command name is null or empty for method '{method.Name}' in type '{declaringType.FullName}'.");
199 return;
200 }
201
202 Add(command, method, attribute);
203 }
204 catch (System.Exception e)
205 {
206 _logger?.Send(
207 MessageType.Exception,
208 $"Exception: {e.GetType().Name}{Environment.NewLine}{e.Message}");
209 throw;
210 }

224 public void Register(string[] unresolvedAssemblyNames, BindingFlags bindingFlags = BindingFlags)
225 {
226 if (unresolvedAssemblyNames is null) return;
227
228 var hashSet = new HashSet<Assembly>();
229
230 foreach (var loadedAssembly in AppDomain.CurrentDomain.GetAssemblies())
231 {
232 hashSet.Add(loadedAssembly);
233 }
234
235 foreach (var unresolvedAssemblyName in unresolvedAssemblyNames)
236 {
237 AssemblyName referencedAssemblyName = null;
238 try
239 {
240 var assembly = Assembly.Load(unresolvedAssemblyName);
241 if (assembly is not null)
242 {
243 var referencedAssembliesName = assembly.GetReferencedAssemblies();
244 foreach (var assemblyName in referencedAssembliesName)
245 {
246 referencedAssemblyName = assemblyName;
247 var referencedAssembly = Assembly.Load(referencedAssemblyName);
248 if (hashSet.Add(referencedAssembly))
249 {
250 Register(referencedAssembly, bindingFlags);
251 }
252 }
253 }
254 else
255 {
256 _logger?.Send(
257 MessageType.Warning,
258 $"Failed to load assembly: {unresolvedAssemblyName}. Assembly.Load returned null.");
259 }
260 }
261 catch (System.Exception e)
262 {
263 _logger?.Send(
264 MessageType.Error,
265 $"Referenced assembly '{referencedAssemblyName}' from assembly '{unresolvedAssemblyName}' could not be loaded.{Environment.NewLine}{e.GetType()}{Environment.NewLine}{e.Message}");
266 throw;
267 }
268 }
