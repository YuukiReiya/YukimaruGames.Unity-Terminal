123 private void OnAutocompleteTriggered()
124 {
125 if (!IsVisible) return;
126
127 var completionResults = _service.Autocomplete(_inputPresenter.InputText);
128 var length = completionResults?.Length ?? 0;
129
130 switch (length)
131 {
132 case 1:
133 _inputPresenter.SetInputField(completionResults![0]);
134 _inputPresenter.SetFocus(true);
135 _inputPresenter.SetMoveCursorToEnd();
136 break;
137 case > 1:
138 const string separator = " ";
139 _service.SystemMessage(string.Join(separator, completionResults!));
140 _inputPresenter.SetMoveCursorToEnd();
141 break;
142 }
143
144 _scrollConfigurator.ScrollToEnd();
145 }
